


## 1.try catch封装

```javascript
// try catch 封装

export async function to(promise) {
  try {
    const response = await promise
    return [response, null] // 返回 [成功结果, null]
  }
  catch (error) {
    return [null, error] // 返回 [null, 错误对象]
  }
}


// useage
async function getList() {
  const [res,err] = await to(getList())
  //......res
  //......error
}

```
